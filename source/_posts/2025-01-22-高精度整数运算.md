# 高精度整数运算

## 目的

为了解决cpp语言整型变量存在范围，太大的整数会导致溢出。

## 存储方式

- 采用vector存储每一位数，一般从低位开始存，方便进位操作。

- 加法、减法、乘法都是从最低位算起的，但是除法得从最高位算起，保存的C数组也是反的，所以返回之前要转置。

## 高精度加法

两个大整数相加。

```cpp
#include<bits/stdc++.h>
using namespace std;

const int N=1e6+10;
vector<int> add(vector<int>&A,vector<int>&B){
    vector<int>C;

    int t=0;
    for(int i=0;i<A.size()||i<B.size();i++){
        if(i<A.size()) t+=A[i];
        if(i<B.size()) t+=B[i];

        C.push_back(t%10);
        t/=10;
    }

    if(t) C.push_back(1);//容易忘记!!!!
    return C;
}

void work(){
    string a,b;
    vector<int> A,B;

    cin>>a>>b;
    for(int i=a.size()-1;i>=0;i--) A.push_back(a[i]-'0');
    for(int i=b.size()-1;i>=0;i--) B.push_back(b[i]-'0');//字符化数字

    auto C=add(A,B);
    for(int i=C.size()-1;i>=0;i--) cout<<C[i];
}

int main(){
    work();
    return 0;
}
```

## 高精度减法

两个大整数相减

```cpp
#include<bits/stdc++.h>
using namespace std;

bool cmp(vector<int>&A,vector<int>&B){
    if(A.size()!=B.size()) return A.size()>B.size();
    for(int i=A.size()-1;i>=0;i--){
        if(A[i]!=B[i]) return A[i]>B[i];
    }

    return true;
}

vector<int> sub(vector<int>&A,vector<int>&B){
    vector<int> C;
    for(int i=0,t=0;i<A.size();i++){
        t=A[i]-t;
        if(i<B.size()) t-=B[i];

        C.push_back((t+10)%10);

        if(t<0) t=1;
        else t=0;
    }

    while(C.size()>1&&C.back()==0) C.pop_back();//去除前导0

    return C;
}

void work(){
    string a,b;
    vector<int>A,B;

    cin>>a>>b;
    for(int i=a.size()-1;i>=0;i--) A.push_back(a[i]-'0');
    for(int i=b.size()-1;i>=0;i--) B.push_back(b[i]-'0');

    if(cmp(A,B)){//要确保大数减小数
        auto C=sub(A,B);
        for(int i=C.size()-1;i>=0;i--) cout<<C[i];
    }else{
        auto C=sub(B,A);
        cout<<"-";//结果的负号单独先输出
        for(int i=C.size()-1;i>=0;i--) cout<<C[i];
    }
}

int main(){
    work();
    return 0;
}
```

## 高精度乘法

一个大整数乘一个小整数。

```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> mul(vector<int>&A,int& n){
    vector<int> C;

    int t=0;
    for(int i=0;i<A.size();i++){
        t=t+A[i]*n;
        C.push_back(t%10);
        t/=10;
    }
    if(t>0) C.push_back(t);

    while (C.size()>1&&C.back()==0) C.pop_back();

    return C;
}

void work(){
    string a;
    cin>>a;

    vector<int>A;
    for(int i=a.size()-1;i>=0;i--) A.push_back(a[i]-'0');

    int n;
    cin>>n;

    auto C=mul(A,n);

    for(int i=C.size()-1;i>=0;i--) cout<<C[i];
    
}

int main(){
    work();
    return 0;
}
```

## 高精度除法

一个大整数除以一个小整数，要返回商和余数。

```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> div(vector<int>&A, int& n,int& r){
    vector<int> C;

    for(int i=A.size()-1;i>=0;i--){//从高位开始除
        r=r*10+A[i];
        C.push_back(r/n);
        r%=n;
    }

    reverse(C.begin(),C.end());//倒置得到答案
    while(C.size()>1&&C.back()==0) C.pop_back();//去除前导0

    return C;
}

void work(){
    string a;
    int n;

    cin>>a>>n;
    vector<int>A;
    
    for(int i=a.size()-1;i>=0;i--) A.push_back(a[i]-'0');

    int r=0;
    auto C=div(A,n,r);

    for(int i=C.size()-1;i>=0;i--) cout<<C[i];
    cout<<endl<<r;
}

int main(){
    work();
    return 0;
}
```
